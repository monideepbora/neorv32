125 //** Construct instruction word (32-bit) for R-type instruction */
126 #define CMD_WORD_R_TYPE(funct7, rs2, rs1, funct3, rd, opcode) \
127  ( (opcode & 0x7f) << 0 ) + \
128  ( (rd & 0x1f) << 7 ) + \
129  ( (rs1 & 0x1f) << 15 ) + \
130  ( (rs2 & 0x1f) << 20 ) + \
131  ( (funct7 & 0x7f) << 25 ) + \
132  ( (funct3 & 0x1f) << 12 )
133 
134 //** Construct instruction word (32-bit) for I-type instruction */
135 #define CMD_WORD_I_TYPE(imm12, rs1_f5, funct3, rd, opcode) \
136  ( (opcode & 0x7f) << 0 ) + \
137  ( (rd & 0x1f) << 7 ) + \
138  ( (rs1_f5 & 0x1f) << 15 ) + \
139  ( (imm12 & 0xfff) << 20 ) + \
140  ( (funct3 & 0x1f) << 12 )
141 
142 //** Construct custom instruction for R-type instruction */
143 #define CUSTOM_INSTR_R_TYPE(funct7, rs2, rs1, funct3, rd, opcode) \
144  asm volatile (".word "STR(CMD_WORD_R_TYPE(funct7, GET_REG_ADDR(rs2), GET_REG_ADDR(rs1), funct3, GET_REG_ADDR(rd), opcode))"\n");
145 
146 //** Construct custom instruction for R1-type instruction (register + 5-bit immediate/function_select) */
147 #define CUSTOM_INSTR_R1_TYPE(funct7, funct5, rs1, funct3, rd, opcode) \
148  asm volatile (".word "STR(CMD_WORD_R_TYPE(funct7, funct5, GET_REG_ADDR(rs1), funct3, GET_REG_ADDR(rd), opcode))"\n");
149 
150 //** Construct custom instruction for I-type instruction */

152  asm volatile (".word "STR(CMD_WORD_I_TYPE(imm12, GET_REG_ADDR(rs1), funct3, GET_REG_ADDR(rd), opcode))"\n");

699 uint32_t riscv_emulate_clz(uint32_t rs1) {
700 
701  uint32_t sreg = rs1;
702  uint32_t cnt = 0;
703 
704  while(1) {
705  if (sreg & 0x80000000UL) {
706  break;
707  }
708  else {
709  sreg <<= 1;
710  cnt++;
711  }
712  }
713 
714  return cnt;
715 }
